Total Transaction:=COUNTROWS('Data')
Total Estimated Income:=SUM(Data[Estimated Income])
Total Properties Owned:=SUM(Data[Properties Owned])
Total Bank Deposits:=SUM([Bank Deposits])
Total Checking Accounts:=SUM([Checking Accounts])
Total Deposit / Checking Accounts:=[Total Bank Deposits] + [Total Checking Accounts]
Transaction Grand Total:=CALCULATE([Total Transaction] , ALL(Data))
Transaaction  Percentage:=DIVIDE([Total Transaction] , [Transaction Grand Total] , 0)
Property Owned Liability:=SUMX(Data,[Properties Owned] * Data[Foreign Currency Account])
Unique Contacts:=DISTINCTCOUNT([Banking Contact])
Unique Gender:=DISTINCTCOUNT(Data[Sex])
SPLY Transactions:=CALCULATE([Total Transaction] , SAMEPERIODLASTYEAR('Calendar'[Date]))
LY Transaction:=CALCULATE([Total Transaction], PREVIOUSYEAR('Calendar'[Date]))
YTD Transactions:=TOTALYTD([Total Transaction],'Calendar'[Date])
QTD Transactions:=TOTALQTD([Total Transaction], 'Calendar'[Date])
